%%%% EXTRACCIO %%%%
\subsection[Extracció]{Extracció d'informació}
\begin{frame}{Procés d'extracció}
    \framesubtitle{Extracció d'informació}
    \begin{block}{Dos tipus de regles d'extracció d'informació:}
        \begin{itemize}
      		\item
            \textit{Reference Wrappers}
            \item
            \textit{Field Wrappers}
        \end{itemize}
    \end{block}
\end{frame}

\begin{frame}{Extracció d'informació}
    \framesubtitle{Reference Wrappers}
    \begin{block}{Característiques:}
        \begin{itemize}
      		\item
            Extreuen referències senceres
            \item
            Només se'n necessita un per cada biblioteca digital
            \item
            Implementats manualment
            \item
            Els resultats solen ser bons
        \end{itemize}
    \end{block}
    \begin{figure}
	    \centering
	    \includegraphics[width=0.75\textwidth]{figures/refextraction:reference-wrappers.pdf}
    \end{figure}
\end{frame}


\begin{frame}{Extracció d'informació}
    \framesubtitle{Field Wrappers}
    \begin{block}{Característiques:}
        \begin{itemize}
      		\item
            Especialitzats en treure un sol camp
            \item
            Se'n necessita un per cada camp i per cada biblioteca
            \item
            Implementats automàticament
        \end{itemize}
    \end{block}
    \begin{figure}
	    \centering
	    \includegraphics[width=\textwidth]{figures/refextraction:field-wrappers.pdf} 
    \end{figure}
\end{frame}


\begin{frame}{Extracció d'informació}
    \framesubtitle{Validació}   
    \begin{block}{Característiques de la validació:}
        \begin{itemize}
      		\item{}
            Camps com ara el títol i autors: es comprova que es troben dins del document PDF
            \item{}
            Camps com ara el número de pàgines o any: es mira que compleixin una expressió regular.
            \item{}
            Cada camp té un pes sore la validesa final
        \end{itemize}
    \end{block}
\end{frame}

\subsection[Tipus de Regles]{Tipus de Regles}
\begin{frame}{Generació de regles}
    \framesubtitle{Tipus de regles}
    \begin{block}{\textit{Path Rule}:}
        \begin{itemize}        
            \item{}
            Localitza l'element del document HTML que conté el valor que busquem
            \item{}
            El patró consisteix en una ruta dins de l'arbre que representa el document
        \end{itemize}
    \end{block}
    \pause
    \begin{block}{\textit{Regex Rule}:}
        \begin{itemize}        
            \item{}
            Localitza el valor dins del contingut de l'element HTML obtingut per una \textit{Path Rule}
            \item{}
            El patró consisteix en una expressió regular per extreure el valor desitjat
        \end{itemize}
    \end{block}
\end{frame}


\begin{frame}{Generació de regles}
    \framesubtitle{Altres regles per a camps multivalor}
    \begin{block}{\textit{Separators Regex Rule}:}
        \begin{itemize}        
            \item{}
            Separa els diferents valors continguts en un mateix element HTML
            \item{}
            El patró consisteix en un conjunt de cadenes de caràcters que separen els valors desitjats
        \end{itemize}
    \end{block}
    \pause
    \begin{block}{\textit{MultiValue Regex Rule}:}
        \begin{itemize}        
            \item{}
            Consisteix en una \textit{regex rule} que actua sobre diferents valors
        \end{itemize}
    \end{block}
    \pause
    \begin{block}{\textit{Person Rule}:}
        \begin{itemize}        
            \item{}
            Separen noms en els camps: \texttt{last\_name}, \texttt{middle\_name} i \texttt{first\_name}
        \end{itemize}
    \end{block}
\end{frame}

\subsection[Avaluació]{Avaluació}
\begin{frame}{Generació de regles}

        \end{itemize}
    \end{block}
\end{frame}
