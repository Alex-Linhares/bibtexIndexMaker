%%%% EXTRACCIO %%%%
\section[Extracció]{Extracció d'informació}
\begin{frame}{Extracció d'informació}
    \begin{block}{Dos mètodes d'extracció:}
        \begin{itemize}
      		\item
            \textit{Reference Wrappers}
            \item
            \textit{Field Wrappers}
        \end{itemize}
    \end{block}
\end{frame}

\begin{frame}{Extracció d'informació}
    \framesubtitle{Reference Wrappers}
    \begin{block}{Característiques:}
        \begin{itemize}
      		\item
            Extreuren referències senceres
            \item
            Només se'n necessita un per cada biblioteca digital
            \item
            Implementats manualment
            \item
            Els resultats solen ser bons
        \end{itemize}
    \end{block}
\end{frame}

\begin{frame}{Extracció d'informació}
    \framesubtitle{Reference Wrappers}

\end{frame}

\begin{frame}{Extracció d'informació}
    \framesubtitle{Field Wrappers}
    \begin{block}{Característiques:}
        \begin{itemize}
      		\item
            Especialitzats en treure un sol camp
            \item
            Se'n necessita un per cada camp i per cada biblioteca
            \item
            Implementats automàticament
        \end{itemize}
    \end{block}
\end{frame}
