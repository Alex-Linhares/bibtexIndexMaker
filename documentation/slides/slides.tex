\begin{frame}{Extracció de referències}
    \framesubtitle{Reference Wrappers}
    \begin{block}{Característiques:}
        \begin{itemize}
      		\item
            Extreuen referències senceres ({\sc Bib}\TeX{})
            \item
            Només se'n necessita un per cada biblioteca digital
            \item
            Implementats manualment
            \item
            Els resultats solen ser bons
        \end{itemize}
    \end{block}
    \begin{figure}
	    \centering
	    \includegraphics[width=0.75\textwidth]{figures/refextraction:reference-wrappers.pdf}
    \end{figure}
\end{frame}


\begin{frame}{Extracció de referències}
    \framesubtitle{Field Wrappers}
    \begin{block}{Característiques:}
        \begin{itemize}
      		\item
            Especialitzats en treure un sol camp
            \item
            Se'n necessita un per cada camp i per cada biblioteca
            \item
            Implementats automàticament
        \end{itemize}
    \end{block}
    \begin{figure}
	    \centering
	    \includegraphics[width=\textwidth]{figures/refextraction:field-wrappers.pdf} 
    \end{figure}
\end{frame}

            \item{}

    \begin{block}{Com ho fem:}

    \end{block}
\end{frame}

%%% REGEX RULES %%%%
\begin{frame}{Generació de regles}
    \framesubtitle{Tipus de regles}
    \begin{block}{\textit{Regex Rule}:}
        \begin{itemize}        
            \item{}
            Localitza valors dins d'una cadena de caràcters.
            \item{}
            El patró és una expressió regular per extreure el valor desitjat
        \end{itemize}
    \end{block}

    \begin{center}
    \begin{tabular}{ll}      
        Text: & \textbf{\texttt{Vol. 27, pp. 1204-1209.}} \\
        Patró: & \textbf{\texttt{(?:.*), pp. (.*).}} \\
        &\\
        Resultat: &\textbf{\texttt{1204-1209}}\\
    \end{tabular}
    \end{center}
\end{frame}

%%% SEPARATORS RULES %%%%
\begin{frame}{Generació de regles}
    \framesubtitle{Altres regles}
    \begin{block}{\textit{Separators Regex Rule}:}
        \begin{itemize}        
            \item{}
            Separa valors continguts en un mateix text
            \item{}
            El patró és un conjunt de cadenes que actuen de separadors
        \end{itemize}
    \end{block}

    \begin{center}
    \begin{tabular}{ll}      
        Text: & \textbf{\texttt{Value A, Value B and Value C}} \\
        Patró: & \textbf{\texttt{``, ''}} i  \textbf{\texttt{`` and ''}}\\
        &\\
        Resultat: &\textbf{\texttt{[Value A, Value B, Value C]}}\\
    \end{tabular}
    \end{center}

\end{frame}

%%%% Altres %%%%
\begin{frame}{Generació de regles}
    \framesubtitle{Altres regles}
    \begin{block}{\textit{MultiValue Regex Rule}:}
        \begin{itemize}        
            \item{}
            \textit{Regex rule} que actua sobre diferents valors
        \end{itemize}
    \end{block}
    \vspace{5mm}
    \pause
    \begin{block}{\textit{Person Rule}:}
        \begin{itemize}        
            \item{}
            Separen noms en els camps: \texttt{last\_name}, \texttt{middle\_name} i \texttt{first\_name}. Sense patró.
        \end{itemize}
    \end{block}
    \begin{center}
    \begin{tabular}{ll}      
        Nom: & \textbf{\texttt{Anders And}} \\
        & \\
        Resultat: &\textbf{\texttt{[``And'', ``'',``Anders'']}}\\
    \end{tabular}
    \end{center}
\end{frame}
